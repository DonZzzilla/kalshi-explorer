PSEUDOCODE

async def scanner_loop():
    markets, cursor = [], None
    while True:
        batch = await kalshi.markets(cursor=cursor, limit=100)
        markets.extend(batch)
        if not batch.next_cursor: break
        cursor = batch.next_cursor

    probs = []
    for m in markets:
        ob = await kalshi.orderbook(m.ticker)
        yes_mid = (ob.yes_best_bid + ob.yes_best_ask) / 2
        spread = ob.yes_best_ask - ob.yes_best_bid
        probs.append({'ticker': m.ticker, 'title': m.title,
                      'implied': yes_mid/100, 'spread': spread})

    liquid = [p for p in probs if p['spread'] < 5 and p['volume'] > 1000]

    for p in liquid:
        p['model_prob'] = your_model.predict(p['title'])
        p['edge'] = p['model_prob'] - p['implied']

    ranked = sorted([p for p in liquid if p['edge'] > 0.02],
                    key=lambda x: x['edge']*x['volume'], reverse=True)

    for bet in ranked[:10]:
        kelly = kelly_criterion(bet['edge'], bet['model_prob'])
        if kelly > 0.01: alert(f"{bet['title']} | {bet['edge']:.1%}")
    return ranked[:10]
